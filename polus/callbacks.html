158class LossSmoothCallback(Callback):
159    """
160    Applies a smooth factor to the loss
161    """
162    def __init__(self, beta = 0.97, output=False):
163        super().__init__()
164        self.beta = beta
165        self.mov_avg = 0
166        self.n = 0
167        self.smooth_loss = 0
168        self.output = output
169    
170    def __maybe_output(self):
171        if self.output:
172            for output in self.coordinator.output_streamers:
173                output.write("smooth loss", self.smooth_loss)
174    
175    @runs_if_root
176    def on_train_batch_end(self, epoch, step, loss):
177        
178        self.n += 1
179        self.mov_avg = self.beta * self.mov_avg + (1 - self.beta) * loss
180        self.smooth_loss = self.mov_avg / (1 - self.beta ** self.n)
181        
182        self.coordinator.shared_dict["smooth_loss"] = self.smooth_loss
183        
184        self.__maybe_output()
185    
186    @runs_if_root
187    def on_epoch_end(self, epoch):
188        self.__maybe_output()

315class EarlyStop(Callback):
316    def __init__(self, 
317                 patience = 3, 
318                 use_smooth_loss = True):
319        super().__init__()
320        self.current_patience = 0
321        self.patience = patience
322        self.last_loss = 1000
323        self.use_smooth_loss = use_smooth_loss
324    
325    @runs_if_root
326    def on_train_begin(self):
327        if self.use_smooth_loss and not self.coordinator.has_callback(LossSmoothCallback):
328            logger.warn("LossSmoothCallback was not found on the coordinator, which is a requirement to use smooth loss. Therefore this call back will use the normal loss")
329            self.use_smooth_loss = False
330            self.loss = []
331        
332    @runs_if_root
333    def on_train_batch_end(self, epoch, step, loss):
334        if not self.use_smooth_loss:
335            self.loss.append(loss)
336    
337    @runs_if_root
338    def on_epoch_end(self, epoch):
339        if self.use_smooth_loss:
340            loss = self.coordinator.shared_dict["smooth_loss"]
341        else:
342            loss = sum(self.loss)/len(self.loss)
343            self.loss = [] # loss per epoch
344        
345        if np.isnan(loss):
346            logger.info(f"The training will stop early since the loss became nan")
347            self.coordinator.trainer.early_stop = True
348            ctx = HPOContext()
349            if ctx.is_hpo_enable():
350                from optuna.exceptions import TrialPruned
351                from optuna.trial import Trial
352                if isinstance(ctx.hpo_backend, Trial): # optuna has backend
353                    raise TrialPruned
354                else:
355                    raise ValueError(f"The loss was nan, but early stop does not know how to end the run with the {ctx.hpo_backend} backend")
356            
357        
358        if self.last_loss < loss:
359            self.current_patience += 1
360            
361        if self.current_patience > self.patience:
362            self.coordinator.trainer.early_stop = True
363            logger.info(f"The training will stop early since the loss did not improve in {self.patience} consecutive epochs")
