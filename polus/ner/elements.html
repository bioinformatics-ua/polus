  12def assert_valid_span(span):
  13    assert isinstance(span, tuple)
  14    assert len(span) == 2
  15    start, end = span
  16    assert isinstance(start, int) and (start >= 0)
  17    assert isinstance(end, int) and (end >= start)

  20def span_overlaps_span(span1, span2):
  21    assert_valid_span(span1)
  22    assert_valid_span(span2)
  23    if (span1[1] <= span2[0]) or (span1[0] >= span2[1]):
  24        return False
  25    else:
  26        return True

  45class Entity:
  46    r"""
  47    An Entity has a textual mention, a span (start and end offsets),
  48    and a type.
  49    
  50    Example (taken from NLM-Chem, dev subset, PMCID 4200806):
  51        text: 'tyrosine'
  52        span: (2914, 2922)
  53        typ: 'Chemical'
  54    
  55    >>> Entity('tyrosine', (2914, 2922), 'Chemical')
  56    
  57    Example (taken from NLM-Chem, train subset, PMCID 5600090):
  58        text: 'MIDA boronate ester'
  59        span: (758, 777)
  60        typ: 'Chemical'
  61    
  62    >>> Entity('MIDA boronate ester', (758, 777), 'Chemical')
  63    """
  64    
  65    def __init__(self, text, span, typ):
  66        assert isinstance(text, str)
  67        assert_valid_span(span)
  68        assert isinstance(typ, str)
  69        
  70        start, end = span
  71        n_characters = end - start
  72        assert len(text) == n_characters
  73        
  74        self.text = text
  75        self.span = span
  76        self.start = start
  77        self.end = end
  78        self.n_characters = n_characters
  79        self.typ = typ
  80    
  81    def __repr__(self):
  82        return 'Entity{}'.format((self.text, self.span, self.typ))
  83    
  84    def __len__(self):
  85        return self.n_characters
  86    
  87    def __eq__(self, other):
  88        r"""
  89        Compare two entities to check if they have:
  90            - the same text
  91            - the same span
  92            - the same type
  93        """
  94        assert type(other) is Entity
  95        if ((self.text == other.text) and
  96            (self.span == other.span) and
  97            (self.typ == other.typ)):
  98            return True
  99        else:
 100            return False
 101    
 102    def __hash__(self):
 103        return hash((self.text, self.span, self.typ))
 104    
 105    def __lt__(self, other):
 106        r"""
 107        This magic method allows to easily sort a list of Entity
 108        objects with the sorted() function.
 109        
 110        Attributes sorting priority:
 111            1. start
 112            2. end
 113            3. typ
 114            4. text
 115        """
 116        assert type(other) is Entity
 117        if self.start < other.start:
 118            return True
 119        elif self.start == other.start:
 120            if self.end < other.end:
 121                return True
 122            elif self.end == other.end:
 123                if self.typ < other.typ:
 124                    return True
 125                elif self.typ == other.typ:
 126                    if self.text < other.text:
 127                        return True
 128        return False
 129    
 130    def to_normalized_entity(self):
 131        return NormalizedEntity(self.text, self.span, self.typ)
 132    
 133    def json(self, i=1):
 134        assert isinstance(i, int)
 135        return {
 136            'id': str(i),
 137            'infons': {
 138                'type': self.typ
 139            },
 140            'locations': [
 141                {
 142                    'length': self.n_characters,
 143                    'offset': self.start
 144                }
 145            ],
 146            'text': self.text
 147        }
 148    
 149    def pretty_json(self, i=1):
 150        return json.dumps(self.json(i), indent=4, sort_keys=True)

 156

 158

 163    >>> es = EntitySet([e1, e2, e3])

 165

 168        if entities is not None:
 169            self.update(entities)
 170

 172        return len(self.entities)
 173

 175        s = ''
 176        for e in self.get(sort=True, make_deepcopy=False):
 177            s += '{}\n'.format(e)

 193        assert type(e) is Entity
 194        self.entities.add(e)
 195

 197        assert isinstance(entities, list) or isinstance(entities, set)
 198        for e in entities:
 199            self.add(e)

 241class NormalizedEntity(Entity):
 242    r"""
 243    A NormalizedEntity inherits from Entity.
 244    
 245    Besides having a textual mention, a span, and a type, additionally
 246    has a list of identifiers (for normalization).
 247    
 248    Note: it has to be a list of identifiers (not a set) because the
 249          order of the identifiers matters. For example, if the entity
 250          text mention refers to three diferent terms, its identifiers
 251          follow their order of appearance (see Example 2 below).
 252    
 253    Example 1 (taken from NLM-Chem, dev subset, PMCID 4200806):
 254        text: 'tyrosine'
 255        span: (2914, 2922)
 256        typ: 'Chemical'
 257        identifiers: ['MESH:D014443']
 258    
 259    >>> NormalizedEntity('tyrosine', (2914, 2922), 'Chemical', ['MESH:D014443'])
 260    
 261    Example 2 (taken from NLM-Chem, train subset, PMCID 5600090):
 262        text: 'MIDA boronate ester'
 263        span: (758, 777)
 264        typ: 'Chemical'
 265        identifiers: ['MESH:C533766', 'MESH:D001897', 'MESH:D004952']
 266    
 267    >>> NormalizedEntity('MIDA boronate ester', (758, 777), 'Chemical', ['MESH:C533766', 'MESH:D001897', 'MESH:D004952'])
 268    
 269    Example 3 (taken from NLM-Chem, train subset, PMCID 4988499)
 270        text: 'cyclic, aromatic, and monoterpenoid enones, enals, and enols'
 271        span: (2793, 2853)
 272        typ: 'Chemical'
 273        identifiers: ['MESH:D007659', 'MESH:D000447', '-']
 274    
 275    >>> NormalizedEntity('cyclic, aromatic, and monoterpenoid enones, enals, and enols', (2793, 2853), 'Chemical', ['MESH:D007659', 'MESH:D000447', '-'])
 276    """
 277    
 278    def __init__(self, text, span, typ, identifiers=None):
 279        super().__init__(text, span, typ)
 280        self.set_identifiers(identifiers)
 281    
 282    def set_identifiers(self, identifiers):
 283        if identifiers is None:
 284            identifiers = list()
 285        
 286        assert isinstance(identifiers, list)
 287        for i in identifiers:
 288            assert isinstance(i, str)
 289        #
 290        # If the identifiers list is empty, it means there are no
 291        # identifiers.
 292        # In this case, add the '-' identifier meaning there is no
 293        # identifier.
 294        #
 295        if len(identifiers) == 0:
 296            identifiers = ['-']
 297        
 298        self.identifiers = identifiers
 299        self.identifiers_str = ','.join(identifiers)
 300    
 301    def __repr__(self):
 302        return 'NormalizedEntity{}'.format((self.text, self.span, self.typ,
 303                                            self.identifiers))
 304    
 305    def __eq__(self, other):
 306        r"""
 307        Compare two normalized entities to check if they have:
 308            - the same text
 309            - the same span
 310            - the same type
 311            - the same identifiers
 312        """
 313        assert type(other) is NormalizedEntity
 314        if ((self.text == other.text) and
 315            (self.span == other.span) and
 316            (self.typ == other.typ) and
 317            (self.identifiers == other.identifiers)):
 318            return True
 319        else:
 320            return False
 321    
 322    def __hash__(self):
 323        return hash((self.text, self.span, self.typ, self.identifiers_str))
 324    
 325    def __lt__(self, other):
 326        r"""
 327        This magic method allows to easily sort a list of
 328        NormalizedEntity objects with the sorted() function.
 329        
 330        Attributes sorting priority:
 331            1. start
 332            2. end
 333            3. typ
 334            4. text
 335            5. identifiers_str
 336        """
 337        assert type(other) is NormalizedEntity
 338        if self.start < other.start:
 339            return True
 340        elif self.start == other.start:
 341            if self.end < other.end:
 342                return True
 343            elif self.end == other.end:
 344                if self.typ < other.typ:
 345                    return True
 346                elif self.typ == other.typ:
 347                    if self.text < other.text:
 348                        return True
 349                    elif self.text == other.text:
 350                        if self.identifiers_str < other.identifiers_str:
 351                            return True
 352        return False
 353    
 354    def to_entity(self):
 355        return Entity(self.text, self.span, self.typ)
 356    
 357    def json(self, i=1):
 358        assert isinstance(i, int)
 359        return {
 360            'id': str(i),
 361            'infons': {
 362                'identifier': self.identifiers_str,
 363                'type': self.typ
 364            },
 365            'locations': [
 366                {
 367                    'length': self.n_characters,
 368                    'offset': self.start
 369                }
 370            ],
 371            'text': self.text
 372        }

 429    r"""

 590class Passage:
 591    r"""
 592    A Passage is initialized with a text, a span (start and end
 593    offsets), a type (abstract, fig_caption, footnote, front,
 594    paragraph, ref, table_caption, title, etc), and a section type
 595    (ABSTRACT, INTRO, METHODS, RESULTS, etc). Note that, in the
 596    NLM-Chem dataset, frequently the section types are undefined.
 597    
 598    At first, a Passage has no annotations. But these can be added
 599    iteratively. Annotations are NormalizedEntity or IndexingIdentifier
 600    objects.
 601    """
 602    
 603    def __init__(self, text, span, typ, section_type):
 604        assert isinstance(text, str)
 605        assert_valid_span(span)
 606        assert isinstance(typ, str)
 607        assert isinstance(section_type, str)
 608        
 609        start, end = span
 610        n_characters = end - start
 611        assert len(text) == n_characters
 612        
 613        self.text = text
 614        self.span = span
 615        self.start = start
 616        self.end = end
 617        self.n_characters = n_characters
 618        self.typ = typ
 619        self.section_type = section_type
 620        self.nes = NormalizedEntitySet()
 621        self.iis = IndexingIdentifierSet()
 622    
 623    def __str__(self):
 624        s = 'Passage {} ({}, {}): {}.'
 625        return s.format(self.span, repr(self.typ),
 626                        repr(self.section_type), repr(self.text))
 627    
 628    def add_entity(self, e):
 629        assert type(e) is NormalizedEntity
 630        #
 631        # Make sure the NormalizedEntity text matches the Passage text
 632        # (in the respective NormalizedEntity span).
 633        #
 634        offset = self.start
 635        e_start = e.start - offset
 636        e_end = e.end - offset
 637        assert e_end <= self.n_characters
 638        assert e.text == self.text[e_start:e_end]
 639        self.nes.add(e)
 640    
 641    def add_entities(self, entities):
 642        for e in entities:
 643            self.add_entity(e)
 644    
 645    def add_indexing_identifier(self, ii):
 646        self.iis.add(ii)
 647    
 648    def add_indexing_identifiers(self, indexing_identifiers):
 649        for ii in indexing_identifiers:
 650            self.add_indexing_identifier(ii)
